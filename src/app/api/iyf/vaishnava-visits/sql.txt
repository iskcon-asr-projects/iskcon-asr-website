-- VAISHNAVA VISITS TABLE
CREATE TABLE IF NOT EXISTS public.iyf_vaishnava_visits (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  title TEXT NOT NULL,
  date TEXT,
  type TEXT NOT NULL DEFAULT 'instagram_link',
  content TEXT NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- ENABLE RLS
ALTER TABLE public.iyf_vaishnava_visits ENABLE ROW LEVEL SECURITY;

-- POLICIES
CREATE POLICY "Public Read Vaishnava Visits" ON public.iyf_vaishnava_visits FOR SELECT USING (true);
CREATE POLICY "Admin All Vaishnava Visits" ON public.iyf_vaishnava_visits FOR ALL USING (true) WITH CHECK (true);
